import React from "react";
import ReactDOM from "react-dom";

function App() {
const date = new Date();
const hours = date.getHours();
let timeOfDay;

if (hours < 12) {
	timeOfDay = "Morning";
} else if (hours >= 12 && hours < 17) {
	timeOfDay ="Afternoon";
} else {
	timeOfDay = "Night";
}

const stylesey = {
color: "#f00";
fontSize: "200px"; 	//convert the units into a string to be recognized
backgroundColor: #0f0;	 		//in JS there are no keywords with hyphen, instead, we use camelCase
};
return (<h1 style={stylesey}> Good {timeOfDay}! </h1>);	//to change styles in JSX, the JSX styles property unlike CSS, it expects it to be a JS object
}

ReactDOM.render(<App />, document.getElementById("root"))

}
